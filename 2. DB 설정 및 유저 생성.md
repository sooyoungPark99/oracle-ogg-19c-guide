# 2. DB 설정 및 유저 생성

> ⚠️ orcl과 clonedb 양쪽 모두 설정이 필요하다. 한쪽만 하면 반드시 에러가 난다.

---

## 1. orcl DB 사전 설정

```bash
. oraenv
# ORACLE_SID = [orcl] ? orcl

sqlplus / as sysdba
```

```sql
-- Archive Log Mode 확인
SELECT LOG_MODE FROM V$DATABASE;

-- NOARCHIVELOG 이면 전환
SHUTDOWN IMMEDIATE;
STARTUP MOUNT;
ALTER DATABASE ARCHIVELOG;
ALTER DATABASE OPEN;

-- 확인 (ARCHIVELOG 이어야 한다)
SELECT LOG_MODE FROM V$DATABASE;

-- Supplemental Logging 활성화
ALTER DATABASE ADD SUPPLEMENTAL LOG DATA;
SELECT SUPPLEMENTAL_LOG_DATA_MIN FROM V$DATABASE;
-- YES 이어야 한다

-- GoldenGate 파라미터 활성화
ALTER SYSTEM SET ENABLE_GOLDENGATE_REPLICATION = TRUE SCOPE=BOTH;
SHOW PARAMETER ENABLE_GOLDENGATE_REPLICATION;

EXIT;
```

## 3. orcl — Extract용 유저 생성

```bash
sqlplus / as sysdba
```

```sql
CREATE USER oggadmin IDENTIFIED BY oracle
  DEFAULT TABLESPACE users
  TEMPORARY TABLESPACE temp
  QUOTA UNLIMITED ON users;

GRANT CREATE SESSION TO oggadmin;
GRANT ALTER SESSION TO oggadmin;
GRANT SELECT ANY DICTIONARY TO oggadmin;
GRANT SELECT ANY TABLE TO oggadmin;
GRANT FLASHBACK ANY TABLE TO oggadmin;
GRANT CREATE TABLE TO oggadmin;
GRANT INSERT ANY TABLE TO oggadmin;
GRANT UPDATE ANY TABLE TO oggadmin;
GRANT DELETE ANY TABLE TO oggadmin;
GRANT EXECUTE ON DBMS_FLASHBACK TO oggadmin;
GRANT SELECT ON SYS.V_$DATABASE TO oggadmin;
GRANT SELECT ON SYS.V_$ARCHIVED_LOG TO oggadmin;
GRANT SELECT ON SYS.V_$LOG TO oggadmin;
GRANT SELECT ON SYS.V_$LOGFILE TO oggadmin;
GRANT SELECT ON SYS.V_$LOGMNR_CONTENTS TO oggadmin;
GRANT LOGMINING TO oggadmin;

-- OGG 19c 필수 추가 권한 (없으면 ADD SCHEMATRANDATA 실패)
GRANT EXECUTE ON SYS.DBMS_XSTREAM_GG_ADM TO oggadmin;
GRANT EXECUTE ON SYS.DBMS_CAPTURE_ADM TO oggadmin;
GRANT EXECUTE ON SYS.DBMS_APPLY_ADM TO oggadmin;
GRANT EXECUTE ON SYS.DBMS_XSTREAM_ADM TO oggadmin;
GRANT SELECT ON SYS.V_$XSTREAM_CAPTURE TO oggadmin;
GRANT SELECT ON SYS.DBA_XSTREAM_RULES TO oggadmin;

EXIT;
```

## 4. clonedb — Replicat용 유저 생성

```bash
sqlplus sys/oracle@clonedb as sysdba
```

```sql
CREATE USER oggadmin IDENTIFIED BY oracle
  DEFAULT TABLESPACE users
  TEMPORARY TABLESPACE temp
  QUOTA UNLIMITED ON users;

-- DBA 권한 부여 (내부 딕셔너리 접근에 필요)
GRANT DBA TO oggadmin;

-- OGG 19c 필수 추가 권한
GRANT EXECUTE ON SYS.DBMS_XSTREAM_GG_ADM TO oggadmin;
GRANT EXECUTE ON SYS.DBMS_CAPTURE_ADM TO oggadmin;
GRANT EXECUTE ON SYS.DBMS_APPLY_ADM TO oggadmin;
GRANT EXECUTE ON SYS.DBMS_XSTREAM_ADM TO oggadmin;
GRANT SELECT ON SYS.V_$XSTREAM_CAPTURE TO oggadmin;
GRANT SELECT ON SYS.DBA_XSTREAM_RULES TO oggadmin;

EXIT;
```
